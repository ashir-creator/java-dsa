//// maximum adjacent difference in array////

class Solution {
    public int maxAdjacentDistance(int[] nums) {
        int maxdiff=0;
        for(int i=0;i<nums.length;i++)
        {
            int next=(i+1)%nums.length;
            int diff=Math.abs(nums[i]-nums[next]);
            maxdiff=Math.max(maxdiff,diff);



         
        }
        return maxdiff;
        

    }
}
